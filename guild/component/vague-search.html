KwmVagueSearch 组件用法 ​

bash
<template>
  <KwmVagueSearch
      url="user/parentsMobileList"
      text="请输入手机号码"
      searchName="mobile"
      valkey="mobile,name"
      label="mobile,name"
      v-model="mobile"
      @confirm="handleSelect"
      :css="{ width: '400px', margin: '20px' }"
      @clear="handleClear"
    />
</template>
